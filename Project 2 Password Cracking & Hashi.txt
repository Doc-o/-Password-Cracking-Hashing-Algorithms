import hashlib
import itertools
import string

# Hash a password
password = "abc"  # original password
hashed = hashlib.sha256(password.encode()).hexdigest()
print("Target Hash:", hashed)

# Brute force cracker (search lowercase letters, 3 chars long)
def crack_sha256(target_hash):
    chars = string.ascii_lowercase
    for guess in itertools.product(chars, repeat=3):
        attempt = ''.join(guess)
        if hashlib.sha256(attempt.encode()).hexdigest() == target_hash:
            return attempt
    return None

found = crack_sha256(hashed)
print("Cracked Password:", found)
